1. 递归, 递归路径每次选择只有两条，+num或者-num
```
class Solution {
  private int[] nums;
  private int target;
  public int findTargetSumWays(int[] nums, int target) {
    this.nums = nums;
    this.target = target;
    return dfs(0, 0);
  }

  private int dfs(int index, int sum) {
    if (index == nums.length) {
      if (sum == target) {
        return 1;
      } else {
        return 0;
      }
    }

    return dfs(index + 1, sum + nums[index]) + dfs(index + 1, sum - nums[index]);
  }
}
```

2. 递归 + 记忆化

```
class Solution {
  private int[] nums;
  private int target;
  public int findTargetSumWays(int[] nums, int taregt) {
    this.nums = nums;
    this.target = target;
    Map<String, Integer> map = new HashMap<>();
    return dfs(0, 0, map);
  }

  private int dfs(int index, int sum, Map<String, Integer> map) {
    if (index == nums.length) {
      if (sum == target) {
        return 1;
      } else {
        return 0;
      }
    }

    String key = index + "-" + sum;
    if (map.containsKey(key)) {
      return map.get(key);
    }

    int res = dfs(index + 1, sum + nums[index]) + dfs(index + 1, sum - nums[index]);
    map.put(key, res);
    return res;
  }
}
```


3. dp,把添加+的树和添加-的树分成两部分
```
+的部分和为S1， -的部分和为S2
S1 + S2 = target
S1 - S2 = nums[0] + nums[1] + ... + nums[n-1] = Sum of array = S
2S1 = target + S
S1 = (target + S) / 2, 背包问题
```

```
class Solution {
  public int findTargetSumWays(int[] nums, int target) {
    int sum = 0;
    for (int num: nums) {
      sum += num;
    }

    int total = sum + target;
    if (total % 2 != 0 || total < 0) {
      return 0;
    }

    total /= 2;
    int[] dp = new int[total+1];
    dp[0] = 1;

    for (int i = 0; i < nums.length; i ++) {
      for (int j = total; j >= nums[i]; j --) {
        dp[j] += dp[j-nums[i]];
      }
    }

    return dp[total];
  }
}
```
